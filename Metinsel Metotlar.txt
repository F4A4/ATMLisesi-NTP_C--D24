public string soruýký(string metin,string deger)
        {
            string sonuc;
            if (deger == "Büyük Yaz")
            {
                sonuc=metin.ToUpper();
            }
            else if(deger=="Küçük Yaz")
            {
                sonuc=metin.ToLower();
            }
            else
            {
                return metin;
            }
            return sonuc;
         }
